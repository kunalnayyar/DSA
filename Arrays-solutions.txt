2. Majority Element
def majorityElement(self, A, N):
        d = {}
        for i in range(len(A)):
            if A[i] not in d:
                d[A[i]] = 1
            else:
                d[A[i]]+=1
        for key,value in zip(d.keys(),d.values()):
            if value > N/2:
                return key
        
        return -1
(Moore's voting algo)
def majorityElement(self, A, N):
        max_ind, count = 0,0
        
        for i in range(len(A)):
            if A[max_ind] == A[i]:
                count+=1
            else:
                count-=1
            if count == -1:
                max_ind = i
                count = 0
                
        count = 0
        for i in A:
            if i == A[max_ind]:
                count+=1
        if count > N/2:
            return A[max_ind]
    
        return -1
